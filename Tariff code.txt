    const searchInput = document.getElementById("searchInput");
    const countryFilter = document.getElementById("countryFilter");
    const sectorFilter = document.getElementById("sectorFilter");
    const resetBtn = document.getElementById("resetBtn");
    const timeline = document.getElementById("timeline");
    const resultsCount = document.getElementById("resultsCount");

    function escapeHtml(value) {
      return String(value)
        .replace(/&/g, "&amp;")
        .replace(/</g, "&lt;")
        .replace(/>/g, "&gt;")
        .replace(/"/g, "&quot;")
        .replace(/'/g, "&#039;");
    }

    function formatDate(dateStr) {
      const date = new Date(dateStr + "T00:00:00");
      if (Number.isNaN(date.getTime())) return dateStr;
      return date.toLocaleDateString("en-CA", {
        year: "numeric",
        month: "short",
        day: "numeric"
      });
    }

    function normalizePieces(value) {
      if (!value || value === "Not specified in source") return [];
      return value
        .split(";")
        .map(part => part.trim())
        .filter(Boolean);
    }

    function getUniqueCountries(events) {
      const set = new Set();
      events.forEach(event => {
        event.country.split(";").map(v => v.trim()).filter(Boolean).forEach(v => set.add(v));
      });
      return Array.from(set).sort();
    }

    function getUniqueSectors(events) {
      const set = new Set();
      events.forEach(event => {
        normalizePieces(event.sectors).forEach(v => set.add(v));
      });
      return Array.from(set).sort((a, b) => a.localeCompare(b));
    }

    function populateFilters() {
      getUniqueCountries(tariffEvents).forEach(country => {
        const option = document.createElement("option");
        option.value = country;
        option.textContent = country;
        countryFilter.appendChild(option);
      });

      getUniqueSectors(tariffEvents).forEach(sector => {
        const option = document.createElement("option");
        option.value = sector;
        option.textContent = sector;
        sectorFilter.appendChild(option);
      });
    }

    function matchesCountry(event, selectedCountry) {
      if (selectedCountry === "all") return true;
      return event.country
        .split(";")
        .map(v => v.trim())
        .includes(selectedCountry);
    }

    function matchesSector(event, selectedSector) {
      if (selectedSector === "all") return true;
      return normalizePieces(event.sectors).includes(selectedSector);
    }

    function matchesSearch(event, query) {
      if (!query) return true;
      const haystack = [
        event.date,
        event.country,
        event.action,
        event.sectors,
        event.statement,
        event.link
      ].join(" ").toLowerCase();

      return haystack.includes(query.toLowerCase());
    }

    function sortNewestFirst(events) {
      return [...events].sort((a, b) => new Date(b.date) - new Date(a.date));
    }

    function renderEvents() {
      const query = searchInput.value.trim();
      const selectedCountry = countryFilter.value;
      const selectedSector = sectorFilter.value;

      const filtered = sortNewestFirst(tariffEvents).filter(event => {
        return (
          matchesCountry(event, selectedCountry) &&
          matchesSector(event, selectedSector) &&
          matchesSearch(event, query)
        );
      });

      resultsCount.textContent = `${filtered.length} event${filtered.length === 1 ? "" : "s"}`;

      if (!filtered.length) {
        timeline.innerHTML = `
          <div class="empty-state">
            No events match your current search or filter settings.
          </div>
        `;
        return;
      }

      timeline.innerHTML = filtered.map(event => {
        const safeDate = escapeHtml(formatDate(event.date));
        const safeCountry = escapeHtml(event.country);
        const safeAction = escapeHtml(event.action);
        const safeSectors = escapeHtml(event.sectors || "Not specified in source");
        const safeStatement = escapeHtml(event.statement || "Not specified in source");

        const linkHtml = event.link && event.link !== "Link not provided in source"
          ? `<a href="${escapeHtml(event.link)}" target="_blank" rel="noopener noreferrer">Open source link</a>`
          : `No source link provided in original file`;

        return `
          <article class="card">
            <div class="meta">
              <span class="date-pill">${safeDate}</span>
              <span class="country-pill">${safeCountry}</span>
            </div>

            <p class="action">${safeAction}</p>

            <div class="details">
              <div class="detail-row">
                <div class="detail-label">Impacted sectors</div>
                <div class="detail-value">${safeSectors}</div>
              </div>

              <div class="detail-row">
                <div class="detail-label">Government statement</div>
                <div class="detail-value">${safeStatement}</div>
              </div>

              <div class="detail-row">
                <div class="detail-label">Reuters / media link</div>
                <div class="detail-value">${linkHtml}</div>
              </div>
            </div>
          </article>
        `;
      }).join("");
    }

    function resetFilters() {
      searchInput.value = "";
      countryFilter.value = "all";
      sectorFilter.value = "all";
      renderEvents();
    }

    populateFilters();
    renderEvents();

    searchInput.addEventListener("input", renderEvents);
    countryFilter.addEventListener("change", renderEvents);
    sectorFilter.addEventListener("change", renderEvents);
    resetBtn.addEventListener("click", resetFilters);
  </script>
